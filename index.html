Sprawdzenie BANA if (data.blocked === true) { document.body.innerHTML = `

🚫 BAN PERMANENTNY 🚫

`; return; } // 2. Aktualizacja licznika monet w UI const coinDisplay = document.getElementById('coinCountHeader'); if (coinDisplay) { coinDisplay.innerText = data.coins || 0; } // 3. Sprawdzenie uprawnień admina (Korona) if (user.uid === ADMIN_UID) { document.body.classList.add('is-admin'); console.log("Witaj Panie Administratorze"); } }); } else { // Logowanie anonimowe, jeśli nie zalogowany signInAnonymously(auth).catch(err => console.error("Błąd logowania:", err)); } }); // --- OBSŁUGA CZATU --- const chatBox = document.getElementById('chat-messages'); function sendMessage(text) { const user = auth.currentUser; if (!user) return; const newMessageRef = push(ref(db, 'messages')); set(newMessageRef, { uid: user.uid, text: text, timestamp: Date.now(), isVip: document.body.classList.contains('vip-status') }); } // Odbieranie wiadomości w czasie rzeczywistym onChildAdded(ref(db, 'messages'), (snapshot) => { const msg = snapshot.val(); renderMessage(msg); }); function renderMessage(msg) { const div = document.createElement('div'); // Wybór stylu bańki let type = 'standard'; if (msg.uid === ADMIN_UID) type = 'admin'; else if (msg.isVip) type = 'vip'; div.className = `msg-bubble msg-${type}`; div.innerHTML = `${msg.text}`; if (chatBox) { chatBox.appendChild(div); chatBox.scrollTop = chatBox.scrollHeight; } } // --- FUNKCJE ADMINISTRACYJNE --- window.adminTools = { // Wyciszanie gracza mutePlayer: (targetUid, minutes) => { update(ref(db, 'users/' + targetUid), { mutedUntil: Date.now() + (minutes * 60000) }); }, // Dodawanie monet giveCoins: (targetUid, amount) => { update(ref(db, 'users/' + targetUid), { coins: increment(amount) }); }, // Banowanie banPlayer: (targetUid) => { update(ref(db, 'users/' + targetUid), { blocked: true }); } };
